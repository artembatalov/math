\newcommand{\faq}[3]{%
  \par\smallskip\noindent
  \textbf{#1.}\enspace\textit{#2}\par
  \nobreak\vspace{0.3ex}\noindent #3\par
  \writespace}

% Proofs: \prf{номер}{заголовок} ... затем \stmt / \dok блоки
